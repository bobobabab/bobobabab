                            # 对象深拷贝

                            export const deepClone = data => {
  var type = getObjType(data)
  var obj
  if (type === 'array') {
    obj = []
  } else if (type === 'object') {
    obj = {}
  } else {
    // 不再具有下一层次
    return data
  }
  if (type === 'array') {
    for (var i = 0, len = data.length; i < len; i++) {
      obj.push(deepClone(data[i]))
    }
  } else if (type === 'object') {
    for (var key in data) {
      obj[key] = deepClone(data[key])
    }
  }
  return obj
}

                            # 格式化时间

                            export const getDate = (data, flag) => {
  let time = ''
  let date = data || new Date(),
    y = date.getFullYear(),
    m = (date.getMonth() + 1).toString().padStart(2, "0"),
    d = date
      .getDate()
      .toString()
      .padStart(2, "0"),
    h = date.getHours().toString()
      .padStart(2, "0"),
    M = date.getMinutes().toString()
      .padStart(2, "0"),
    S = date.getSeconds().toString()
      .padStart(2, "0");


  if (flag) {
    time = `${y}-${m}-${d} ${h}:${M}:${S}`
  } else {
    time = `${y}/${m}/${d}`
  }
  return time;
}

                            # 复制方法

                            export const copyUrl = (text) => {
  let input = document.createElement("input");
  input.value = text;
  document.body.appendChild(input);
  input.select();
  input.setSelectionRange(0, input.value.length)

  if (document.execCommand) {
    document.execCommand('Copy');
    Message.success('复制成功');
  } else {
    Message.error('复制失败');
  }

  document.body.removeChild(input);

}

                            # 函数节流

                            export function throttle(fn, delay) {
  var lastTime;
  var timer;
  var delay = delay || 200;
  return function () {
    var args = arguments;
    // 记录当前函数触发的时间
    var nowTime = Date.now();
    if (lastTime && nowTime - lastTime < delay) {
      clearTimeout(timer);
      timer = setTimeout(() => {
        // 记录上一次函数触发的时间
        lastTime = nowTime;
        // 修正this指向问题
        fn.apply(this, args);
      }, delay);
    } else {

      fn.apply(this, args);
      lastTime = nowTime;
    }
  }
}

                            # 函数防抖

                            export function debounce(func, wait = 1000, immediate = true) {
  let timer = null
  let flag = true
  return function (...args) {
    if (timer) clearTimeout(timer)
    if (flag) {
      flag = false
      if (immediate) {
        func.apply(this, args)
      }
    }
    timer = setTimeout(() => {
      flag = true
      if (!immediate) {
        func.apply(this, args)
      }
    }, wait)
  }
}

                            # 生成随机 len 位数

                            export const randomLenNum = (len, date) => {
  let random = ''
  random = Math.ceil(Math.random() * 100000000000000).toString().substr(0, len || 4)
  if (date) random = random + Date.now()
  return random
}

                            # 时间戳转为固定格式时间

                            export const timeTamp2FormateTime = function (val) {
  const h = parseInt(val / 3600);
  const m = parseInt(val / 60) % 60;
  const s = val % 60;
  return `${h > 9 ? h : `0${h}`}:${m > 9 ? m : `0${m}`}:${
    s > 9 ? s : `0${s}`
    }`;
}

                            # 判断设备类型

                            export const checkClinetType = function() {
  //获取请求头中的userAgent
  const u = navigator.userAgent;
  //是否是android浏览器
  const isAndroid = u.indexOf('Android') > -1;
  //是否是ios浏览器
  const isIos = !!u.match(/\(i[^;]+;( U;)? CPU.+Mac OS X/);
  let type = '';
  if (isAndroid) {
    type = "android";
  } else if (isIos) {
    type = 'ios';
  } else {
    type = 'pc';
  }
  return type
}

                            # xss 预防方式

                            export const xssCheck = function entities(s) {
    let e = {
        '"': '&quot;',
        '&': '&amp;',
        '<': '&lt;',
        '>': '&gt;'
    }
    return s.replace(/["<>&]/g, m => {
        return e[m]
    })
}

                            # 平滑滚动至顶部

                            export const scrollToTop = () => {
  const c = document.documentElement.scrollTop || document.body.scrollTop;
  if (c > 0) {
    window.requestAnimationFrame(scrollToTop);
    window.scrollTo(0, c - c / 8);
  }
};

                            # 缓存函数

                            export const memorize = function(fn) {
  const cache = {}
  return function(...args) {
    const _args = JSON.stringify(args)
    return cache[_args] || (cache[_args] = fn.apply(fn, args))
  }
}
